Package cleanr checks your code for some of the most common layout flaws.

library("cleanr")
set_cleanr_options(reset = TRUE)
file_name <- system.file("source", "R", "utils.R", package = "cleanr")

message("If a file passes all checks, cleanr returns invisibly TRUE")
## If a file passes all checks, cleanr returns invisibly TRUE
print(suppressWarnings(check_file(file_name)))
## [1] TRUE
message("Show current options:")
## Show current options:
get_cleanr_options(flatten_list = TRUE)
##         max_width        max_length         max_lines max_lines_of_code 
##                80               300                65                50 
##     max_arguments max_nesting_depth    max_line_width 
##                 5                 3                80
message("Change an option to procude conditions:")
## Change an option to procude conditions:
set_cleanr_options(max_width = get_cleanr_options("max_width") - 1)

message("Catch cleanr conditions:")
## Catch cleanr conditions:
print(tryCatch(suppressWarnings(check_file(file_name)),
               cleanr = function(e) return(e)))
